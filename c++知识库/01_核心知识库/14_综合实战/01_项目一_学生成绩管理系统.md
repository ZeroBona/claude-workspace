# 项目一：学生成绩管理系统 📊

---

## 项目要求

开发一个完整的学生成绩管理系统，具备以下功能：

1. 添加学生信息（姓名、学号、各科成绩）
2. 显示所有学生信息
3. 按总成绩排名
4. 查询单个学生信息
5. 统计班级平均分、最高分、最低分
6. 保存数据到文件 / 从文件加载数据

## 涉及的知识点

| 知识点 | 用在哪儿 |
|-------|---------|
| 结构体 | 定义学生数据类型 |
| 数组/vector | 存储多个学生 |
| 循环 | 遍历、菜单反复显示 |
| 分支/switch | 菜单选择 |
| 函数 | 模块化各功能 |
| 排序 | 成绩排名 |
| 文件操作 | 保存和加载数据 |

---

## 参考代码结构

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <fstream>
#include <iomanip>
using namespace std;

struct Student {
    string name;
    string id;
    double chinese;
    double math;
    double english;

    double total() const {
        return chinese + math + english;
    }

    double avg() const {
        return total() / 3;
    }
};

// 函数声明
void showMenu();
void addStudent(vector<Student> &students);
void showAll(const vector<Student> &students);
void rankByTotal(vector<Student> &students);
void searchStudent(const vector<Student> &students);
void showStatistics(const vector<Student> &students);
void saveToFile(const vector<Student> &students);
void loadFromFile(vector<Student> &students);

int main() {
    vector<Student> students;
    int choice;

    do {
        showMenu();
        cin >> choice;

        switch (choice) {
            case 1: addStudent(students); break;
            case 2: showAll(students); break;
            case 3: rankByTotal(students); break;
            case 4: searchStudent(students); break;
            case 5: showStatistics(students); break;
            case 6: saveToFile(students); break;
            case 7: loadFromFile(students); break;
            case 0: cout << "谢谢使用！" << endl; break;
            default: cout << "输入错误！" << endl;
        }

        cout << endl;
    } while (choice != 0);

    return 0;
}

void showMenu() {
    cout << "╔════════════════════════╗" << endl;
    cout << "║  学生成绩管理系统       ║" << endl;
    cout << "╠════════════════════════╣" << endl;
    cout << "║ 1. 添加学生            ║" << endl;
    cout << "║ 2. 显示所有学生        ║" << endl;
    cout << "║ 3. 按总分排名          ║" << endl;
    cout << "║ 4. 查询学生            ║" << endl;
    cout << "║ 5. 统计信息            ║" << endl;
    cout << "║ 6. 保存到文件          ║" << endl;
    cout << "║ 7. 从文件加载          ║" << endl;
    cout << "║ 0. 退出               ║" << endl;
    cout << "╚════════════════════════╝" << endl;
    cout << "请选择：";
}

void addStudent(vector<Student> &students) {
    Student s;
    cout << "请输入姓名：";
    cin >> s.name;
    cout << "请输入学号：";
    cin >> s.id;
    cout << "请输入语文成绩：";
    cin >> s.chinese;
    cout << "请输入数学成绩：";
    cin >> s.math;
    cout << "请输入英语成绩：";
    cin >> s.english;
    students.push_back(s);
    cout << "✅ 添加成功！" << endl;
}

void showAll(const vector<Student> &students) {
    if (students.empty()) {
        cout << "暂无学生数据" << endl;
        return;
    }
    cout << "姓名\t学号\t语文\t数学\t英语\t总分\t平均分" << endl;
    cout << "----------------------------------------" << endl;
    for (const auto &s : students) {
        cout << s.name << "\t" << s.id << "\t"
             << s.chinese << "\t" << s.math << "\t"
             << s.english << "\t" << s.total()
             << "\t" << fixed << setprecision(1) << s.avg() << endl;
    }
}

void rankByTotal(vector<Student> &students) {
    sort(students.begin(), students.end(),
         [](const Student &a, const Student &b) {
             return a.total() > b.total();
         });

    cout << "排名\t姓名\t总分" << endl;
    cout << "----------------" << endl;
    for (size_t i = 0; i < students.size(); i++) {
        cout << i + 1 << "\t" << students[i].name
             << "\t" << students[i].total() << endl;
    }
}

void searchStudent(const vector<Student> &students) {
    string keyword;
    cout << "请输入要查询的姓名或学号：";
    cin >> keyword;

    bool found = false;
    for (const auto &s : students) {
        if (s.name == keyword || s.id == keyword) {
            cout << "姓名：" << s.name << " 学号：" << s.id << endl;
            cout << "语文：" << s.chinese << " 数学：" << s.math
                 << " 英语：" << s.english << endl;
            cout << "总分：" << s.total() << " 平均分：" << s.avg() << endl;
            found = true;
        }
    }
    if (!found) cout << "未找到该学生" << endl;
}

void showStatistics(const vector<Student> &students) {
    if (students.empty()) {
        cout << "暂无数据" << endl;
        return;
    }

    double sum = 0, maxScore = 0, minScore = 999;
    for (const auto &s : students) {
        sum += s.total();
        if (s.total() > maxScore) maxScore = s.total();
        if (s.total() < minScore) minScore = s.total();
    }

    cout << "学生人数：" << students.size() << endl;
    cout << "班级总分平均：" << sum / students.size() << endl;
    cout << "最高总分：" << maxScore << endl;
    cout << "最低总分：" << minScore << endl;
}

void saveToFile(const vector<Student> &students) {
    ofstream fout("students.txt");
    for (const auto &s : students) {
        fout << s.name << " " << s.id << " "
             << s.chinese << " " << s.math << " " << s.english << endl;
    }
    fout.close();
    cout << "✅ 数据已保存到 students.txt" << endl;
}

void loadFromFile(vector<Student> &students) {
    ifstream fin("students.txt");
    if (!fin.is_open()) {
        cout << "❌ 文件打开失败" << endl;
        return;
    }

    students.clear();
    Student s;
    while (fin >> s.name >> s.id >> s.chinese >> s.math >> s.english) {
        students.push_back(s);
    }
    fin.close();
    cout << "✅ 已加载 " << students.size() << " 条记录" << endl;
}
```
