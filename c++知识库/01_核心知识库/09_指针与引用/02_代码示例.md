## 示例 1：指针基础 🎯

**文件**：[code/01_pointer_basic.cpp](code/01_pointer_basic.cpp)

```cpp
#include <iostream>
using namespace std;

int main() {
    int x = 10;
    int *p = &x;

    cout << "x 的值：" << x << endl;
    cout << "x 的地址：" << &x << endl;
    cout << "p 的值（存的地址）：" << p << endl;
    cout << "p 指向的值：" << *p << endl;

    *p = 20;
    cout << "修改后 x 的值：" << x << endl;

    return 0;
}
```

**运行效果**：
```
x 的值：10
x 的地址：0x61fe14
p 的值（存的地址）：0x61fe14
p 指向的值：10
修改后 x 的值：20
```

## 示例 2：指针交换 🔄

**文件**：[code/02_pointer_swap.cpp](code/02_pointer_swap.cpp)

```cpp
#include <iostream>
using namespace std;

void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

int main() {
    int x = 10, y = 20;
    cout << "交换前：x=" << x << ", y=" << y << endl;
    swap(&x, &y);
    cout << "交换后：x=" << x << ", y=" << y << endl;
    return 0;
}
```

**运行效果**：
```
交换前：x=10, y=20
交换后：x=20, y=10
```

## 示例 3：引用交换 🔗

**文件**：[code/03_reference_swap.cpp](code/03_reference_swap.cpp)

```cpp
#include <iostream>
using namespace std;

void swap(int &a, int &b) {
    int temp = a;
    a = b;
    b = temp;
}

int main() {
    int x = 10, y = 20;
    cout << "交换前：x=" << x << ", y=" << y << endl;
    swap(x, y);
    cout << "交换后：x=" << x << ", y=" << y << endl;
    return 0;
}
```

**运行效果**：
```
交换前：x=10, y=20
交换后：x=20, y=10
```

## 示例 4：指针遍历数组 📋

**文件**：[code/04_pointer_array.cpp](code/04_pointer_array.cpp)

```cpp
#include <iostream>
using namespace std;

int main() {
    int arr[] = {10, 20, 30, 40, 50};
    int *p = arr;

    for (int i = 0; i < 5; i++) {
        cout << "arr[" << i << "] = " << *p << " (地址:" << p << ")" << endl;
        p++;
    }

    return 0;
}
```

**运行效果**：
```
arr[0] = 10 (地址:0x61fdf0)
arr[1] = 20 (地址:0x61fdf4)
arr[2] = 30 (地址:0x61fdf8)
arr[3] = 40 (地址:0x61fdfc)
arr[4] = 50 (地址:0x61fe00)
```

## 示例 5：空指针与指针安全 🛡️

**文件**：[code/05_null_pointer.cpp](code/05_null_pointer.cpp)

```cpp
#include <iostream>
using namespace std;

int main() {
    int *p = nullptr;
    int x = 42;
    p = &x;

    if (p != nullptr) {
        cout << "p 指向的值：" << *p << endl;
    }

    int **pp = &p;
    cout << "指针的指针：" << **pp << endl;

    return 0;
}
```

**运行效果**：
```
p 指向的值：42
指针的指针：42
```
