## 示例 1：vector 基础操作 🎯

**文件**：[code/01_vector_demo.cpp](code/01_vector_demo.cpp)

```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    vector<int> scores;
    scores.push_back(85);
    scores.push_back(92);
    scores.push_back(78);
    scores.push_back(96);

    cout << "元素个数：" << scores.size() << endl;
    cout << "第一个：" << scores.front() << " 最后一个：" << scores.back() << endl;

    cout << "所有成绩：";
    for (int i = 0; i < scores.size(); i++) cout << scores[i] << " ";
    cout << endl;

    scores.pop_back();
    cout << "删除后个数：" << scores.size() << endl;
    return 0;
}
```

## 示例 2：sort 排序 📊

**文件**：[code/02_sort_demo.cpp](code/02_sort_demo.cpp)

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    vector<int> v = {5, 2, 8, 1, 9, 3, 7};
    cout << "排序前：";
    for (int x : v) cout << x << " ";
    cout << endl;

    sort(v.begin(), v.end());
    cout << "从小到大：";
    for (int x : v) cout << x << " ";
    cout << endl;

    sort(v.begin(), v.end(), greater<int>());
    cout << "从大到小：";
    for (int x : v) cout << x << " ";
    cout << endl;
    return 0;
}
```

## 示例 3：栈和队列 🏗️

**文件**：[code/03_stack_queue.cpp](code/03_stack_queue.cpp)

```cpp
#include <iostream>
#include <stack>
#include <queue>
using namespace std;

int main() {
    stack<int> s;
    s.push(1); s.push(2); s.push(3);
    cout << "栈（后进先出）：";
    while (!s.empty()) { cout << s.top() << " "; s.pop(); }
    cout << endl;

    queue<int> q;
    q.push(1); q.push(2); q.push(3);
    cout << "队列（先进先出）：";
    while (!q.empty()) { cout << q.front() << " "; q.pop(); }
    cout << endl;
    return 0;
}
```

**运行效果**：
```
栈（后进先出）：3 2 1
队列（先进先出）：1 2 3
```

## 示例 4：vector 高级操作 🔧

**文件**：[code/04_vector_operations.cpp](code/04_vector_operations.cpp)

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    vector<int> nums = {5, 2, 8, 1, 9};

    cout << "原始数据：";
    for (int x : nums) cout << x << " ";
    cout << endl;

    sort(nums.begin(), nums.end());
    cout << "从小到大：";
    for (int x : nums) cout << x << " ";
    cout << endl;

    reverse(nums.begin(), nums.end());
    cout << "反转后：";
    for (int x : nums) cout << x << " ";
    cout << endl;

    cout << "最大值：" << *max_element(nums.begin(), nums.end());
    cout << " 最小值：" << *min_element(nums.begin(), nums.end()) << endl;
    return 0;
}
```

## 示例 5：栈和队列的实战应用 🎯

**文件**：[code/05_stack_queue_more.cpp](code/05_stack_queue_more.cpp)

```cpp
#include <iostream>
#include <stack>
#include <queue>
using namespace std;

int main() {
    cout << "=== 栈：进制转换 ===" << endl;
    int n = 42;
    stack<int> s;
    while (n > 0) { s.push(n % 2); n /= 2; }
    cout << "42 的二进制：";
    while (!s.empty()) { cout << s.top(); s.pop(); }
    cout << endl << endl;

    cout << "=== 队列：处理任务 ===" << endl;
    queue<string> q;
    q.push("任务1"); q.push("任务2"); q.push("任务3");
    while (!q.empty()) {
        cout << "处理：" << q.front() << endl;
        q.pop();
    }
    return 0;
}
```
