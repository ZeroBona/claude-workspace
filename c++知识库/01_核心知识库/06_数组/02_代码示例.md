## 示例 1：数组基础操作 🎯

**文件**：[code/01_array_basic.cpp](code/01_array_basic.cpp)

```cpp
#include <iostream>
using namespace std;

int main() {
    // 定义并初始化数组
    int arr[5] = {10, 20, 30, 40, 50};

    // 访问元素
    cout << "第一个元素：" << arr[0] << endl;
    cout << "第三个元素：" << arr[2] << endl;

    // 修改元素
    arr[1] = 99;
    cout << "修改后第二个元素：" << arr[1] << endl;

    // 用for遍历
    cout << "数组全部元素：";
    for (int i = 0; i < 5; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;

    // 从键盘输入
    int scores[3];
    cout << "请输入3个成绩：";
    for (int i = 0; i < 3; i++) {
        cin >> scores[i];
    }
    cout << "你输入的成绩：";
    for (int i = 0; i < 3; i++) {
        cout << scores[i] << " ";
    }
    cout << endl;

    return 0;
}
```

**运行效果**：
```
第一个元素：10
第三个元素：30
修改后第二个元素：99
数组全部元素：10 99 30 40 50
请输入3个成绩：88 92 76
你输入的成绩：88 92 76
```

## 示例 2：求最高分和最低分 🏆

**文件**：[code/02_min_max.cpp](code/02_min_max.cpp)

```cpp
#include <iostream>
using namespace std;

int main() {
    const int N = 5;
    int scores[N];

    cout << "请输入" << N << "个成绩：";
    for (int i = 0; i < N; i++) {
        cin >> scores[i];
    }

    int maxScore = scores[0];
    int minScore = scores[0];

    for (int i = 1; i < N; i++) {
        if (scores[i] > maxScore) maxScore = scores[i];
        if (scores[i] < minScore) minScore = scores[i];
    }

    cout << "最高分：" << maxScore << endl;
    cout << "最低分：" << minScore << endl;

    return 0;
}
```

**运行效果**：
```
请输入5个成绩：85 92 78 96 88
最高分：96
最低分：78
```

## 示例 3：线性查找 🔍

**文件**：[code/03_linear_search.cpp](code/03_linear_search.cpp)

```cpp
#include <iostream>
using namespace std;

int main() {
    int arr[] = {4, 7, 2, 9, 5, 1, 8, 3};
    int n = sizeof(arr) / sizeof(arr[0]);

    int target;
    cout << "请输入要查找的数字：";
    cin >> target;

    int found = -1;
    for (int i = 0; i < n; i++) {
        if (arr[i] == target) {
            found = i;
            break;
        }
    }

    if (found != -1) {
        cout << target << " 找到了！位置在：" << found << endl;
    } else {
        cout << target << " 不在数组中" << endl;
    }

    return 0;
}
```

**运行效果**：
```
请输入要查找的数字：9
9 找到了！位置在：3
```

## 示例 4：冒泡排序（从小到大）📊

**文件**：[code/04_bubble_sort.cpp](code/04_bubble_sort.cpp)

```cpp
#include <iostream>
using namespace std;

int main() {
    int arr[] = {64, 34, 25, 12, 22, 11, 90};
    int n = sizeof(arr) / sizeof(arr[0]);

    cout << "排序前：";
    for (int i = 0; i < n; i++) cout << arr[i] << " ";
    cout << endl;

    // 冒泡排序
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - 1 - i; j++) {
            if (arr[j] > arr[j + 1]) {
                // 交换 arr[j] 和 arr[j+1]
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }

    cout << "排序后：";
    for (int i = 0; i < n; i++) cout << arr[i] << " ";
    cout << endl;

    return 0;
}
```

**运行效果**：
```
排序前：64 34 25 12 22 11 90
排序后：11 12 22 25 34 64 90
```

## 示例 5：二维数组——矩阵转置 🔄

**文件**：[code/05_matrix.cpp](code/05_matrix.cpp)

```cpp
#include <iostream>
using namespace std;

int main() {
    const int ROWS = 2;
    const int COLS = 3;

    int matrix[ROWS][COLS] = {
        {1, 2, 3},
        {4, 5, 6}
    };

    cout << "原矩阵：" << endl;
    for (int i = 0; i < ROWS; i++) {
        for (int j = 0; j < COLS; j++) {
            cout << matrix[i][j] << "\t";
        }
        cout << endl;
    }

    cout << "\n转置后：" << endl;
    for (int j = 0; j < COLS; j++) {
        for (int i = 0; i < ROWS; i++) {
            cout << matrix[i][j] << "\t";
        }
        cout << endl;
    }

    return 0;
}
```

**运行效果**：
```
原矩阵：
1   2   3
4   5   6

转置后：
1   4
2   5
3   6
```
