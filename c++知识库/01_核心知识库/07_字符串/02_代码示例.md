## 示例 1：字符串基本操作 🎯

**文件**：[code/01_string_basic.cpp](code/01_string_basic.cpp)

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string s1 = "Hello";
    string s2 = "World";

    // 拼接
    string s3 = s1 + " " + s2;
    cout << "s3 = " << s3 << endl;

    // 长度
    cout << "s3 长度：" << s3.length() << endl;

    // 下标访问
    cout << "第一个字符：" << s3[0] << endl;
    cout << "最后一个字符：" << s3[s3.length()-1] << endl;

    // 修改
    s3[6] = 'w';
    cout << "修改后：" << s3 << endl;

    // 比较
    cout << "s1 == s2 ? " << (s1 == s2 ? "相等" : "不相等") << endl;

    return 0;
}
```

**运行效果**：
```
s3 = Hello World
s3 长度：11
第一个字符：H
最后一个字符：d
修改后：Hello world
s1 == s2 ? 不相等
```

## 示例 2：统计字符类型 📊

**文件**：[code/02_count_chars.cpp](code/02_count_chars.cpp)

```cpp
#include <iostream>
#include <string>
#include <cctype>
using namespace std;

int main() {
    string s;
    cout << "请输入一行文字：";
    getline(cin, s);

    int letters = 0, digits = 0, spaces = 0, others = 0;

    for (int i = 0; i < s.length(); i++) {
        if (isalpha(s[i])) letters++;
        else if (isdigit(s[i])) digits++;
        else if (isspace(s[i])) spaces++;
        else others++;
    }

    cout << "字母：" << letters << "个" << endl;
    cout << "数字：" << digits << "个" << endl;
    cout << "空格：" << spaces << "个" << endl;
    cout << "其他：" << others << "个" << endl;

    return 0;
}
```

**运行效果**：
```
请输入一行文字：Hello 123 !@#
字母：5个
数字：3个
空格：2个
其他：3个
```

## 示例 3：回文判断 🔄

**文件**：[code/03_palindrome.cpp](code/03_palindrome.cpp)

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string s;
    cout << "请输入一个字符串：";
    cin >> s;

    bool isPalindrome = true;

    for (int i = 0; i < s.length() / 2; i++) {
        if (s[i] != s[s.length() - 1 - i]) {
            isPalindrome = false;
            break;
        }
    }

    if (isPalindrome) {
        cout << s << " 是回文 ✅" << endl;
    } else {
        cout << s << " 不是回文 ❌" << endl;
    }

    return 0;
}
```

**运行效果**：
```
请输入一个字符串：level
level 是回文 ✅
```

## 示例 4：字符串替换 🔧

**文件**：[code/04_string_replace.cpp](code/04_string_replace.cpp)

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string s;
    char oldChar, newChar;

    cout << "请输入字符串：";
    getline(cin, s);
    cout << "要替换的字符：";
    cin >> oldChar;
    cout << "替换为：";
    cin >> newChar;

    for (int i = 0; i < s.length(); i++) {
        if (s[i] == oldChar) {
            s[i] = newChar;
        }
    }

    cout << "替换后：" << s << endl;

    return 0;
}
```

**运行效果**：
```
请输入字符串：hello world
要替换的字符：o
替换为：0
替换后：hell0 w0rld
```

## 示例 5：单词计数 📝

**文件**：[code/05_word_count.cpp](code/05_word_count.cpp)

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string s;
    cout << "请输入一句话：";
    getline(cin, s);

    int count = 0;
    bool inWord = false;

    for (int i = 0; i < s.length(); i++) {
        if (s[i] != ' ' && !inWord) {
            count++;          // 遇到新单词的开头
            inWord = true;
        } else if (s[i] == ' ') {
            inWord = false;   // 遇到空格，单词结束
        }
    }

    cout << "单词数量：" << count << endl;

    return 0;
}
```

**运行效果**：
```
请输入一句话：I love C++ programming
单词数量：4
```
