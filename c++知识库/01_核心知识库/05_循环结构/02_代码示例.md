## 示例 1：while 循环 — 1到100求和 🧮

**文件**：[code/01_while_sum.cpp](code/01_while_sum.cpp)

```cpp
#include <iostream>
using namespace std;

int main() {
    int i = 1;
    int sum = 0;

    while (i <= 100) {
        sum += i;
        i++;
    }

    cout << "1 + 2 + ... + 100 = " << sum << endl;

    return 0;
}
```

**运行效果**：
```
1 + 2 + ... + 100 = 5050
```

## 示例 2：do-while — 猜数字游戏 🎯

**文件**：[code/02_do_while_guess.cpp](code/02_do_while_guess.cpp)

```cpp
#include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

int main() {
    srand(time(0));
    int target = rand() % 100 + 1;   // 1~100 随机数
    int guess;

    cout << "=== 猜数字游戏 ===" << endl;
    cout << "我心里想了一个 1~100 的数字" << endl;

    do {
        cout << "猜一个数：";
        cin >> guess;

        if (guess > target) {
            cout << "太大了！" << endl;
        } else if (guess < target) {
            cout << "太小了！" << endl;
        }
    } while (guess != target);

    cout << "🎉 猜对了！太棒了！" << endl;

    return 0;
}
```

**运行效果**：
```
=== 猜数字游戏 ===
我心里想了一个 1~100 的数字
猜一个数：50
太大了！
猜一个数：25
太小了！
猜一个数：37
太大了！
猜一个数：31
🎉 猜对了！太棒了！
```

## 示例 3：for 循环 — 求奇数和与偶数和 🎯

**文件**：[code/03_for_sum.cpp](code/03_for_sum.cpp)

```cpp
#include <iostream>
using namespace std;

int main() {
    int oddSum = 0;   // 奇数和
    int evenSum = 0;  // 偶数和

    for (int i = 1; i <= 100; i++) {
        if (i % 2 == 0) {
            evenSum += i;   // 偶数加到 evenSum
        } else {
            oddSum += i;    // 奇数加到 oddSum
        }
    }

    cout << "1~100 奇数和 = " << oddSum << endl;
    cout << "1~100 偶数和 = " << evenSum << endl;
    cout << "总和 = " << oddSum + evenSum << endl;

    return 0;
}
```

**运行效果**：
```
1~100 奇数和 = 2500
1~100 偶数和 = 2550
总和 = 5050
```

## 示例 4：break 和 continue 🚦

**文件**：[code/04_break_continue.cpp](code/04_break_continue.cpp)

```cpp
#include <iostream>
using namespace std;

int main() {
    // break 演示：找到第一个能被 7 整除的数
    cout << "1到20中，第一个能被7整除的数是：";
    for (int i = 1; i <= 20; i++) {
        if (i % 7 == 0) {
            cout << i << endl;
            break;   // 找到了，提前结束循环
        }
    }

    // continue 演示：输出 1~10，跳过 5
    cout << "1~10，跳过5：";
    for (int i = 1; i <= 10; i++) {
        if (i == 5) {
            continue;   // 遇到5，跳过这次循环
        }
        cout << i << " ";
    }
    cout << endl;

    return 0;
}
```

**运行效果**：
```
1到20中，第一个能被7整除的数是：7
1~10，跳过5：1 2 3 4 6 7 8 9 10
```

## 示例 5：嵌套循环 — 打印图案 🎨

**文件**：[code/05_patterns.cpp](code/05_patterns.cpp)

```cpp
#include <iostream>
using namespace std;

int main() {
    int n;

    cout << "请输入图形大小：";
    cin >> n;

    // 矩形
    cout << "\\n矩形：" << endl;
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) {
            cout << "* ";
        }
        cout << endl;
    }

    // 直角三角形
    cout << "\\n直角三角形：" << endl;
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= i; j++) {
            cout << "* ";
        }
        cout << endl;
    }

    // 九九乘法表
    cout << "\\n九九乘法表：" << endl;
    for (int i = 1; i <= 9; i++) {
        for (int j = 1; j <= i; j++) {
            cout << j << "×" << i << "=" << i * j << "\\t";
        }
        cout << endl;
    }

    return 0;
}
```

**运行效果（n=5）**：
```
请输入图形大小：5

矩形：
* * * * *
* * * * *
* * * * *
* * * * *
* * * * *

直角三角形：
*
* *
* * *
* * * *
* * * * *

九九乘法表：
1×1=1
1×2=2  2×2=4
...
```
